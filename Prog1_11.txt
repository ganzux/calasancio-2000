/* programa que convierte grados sexagesimales introducidos por el usuario en radianes */

#include <stdio.h>

{
	float grados,radianes;

		printf("Dame los grados: ");

				radianes=grados*3.1415827/180;

		printf("\n\n%f° son %f rad.",grados,radianes);

}
